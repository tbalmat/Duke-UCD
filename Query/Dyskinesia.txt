select d1.conceptid, d1.term, rd.term, d2.term
from   description d1 join relationship r1 on d1.conceptid=r1.sourceid
       join description rd on r1.typeid=rd.conceptid
       join description d2 on r1.destinationid=d2.conceptid
where  d1.dukeselection=1 and d1.term='dyskinesia (finding)' --'motor dysfunction (finding)'
       --and r1.dukeselection=1 and rd.dukeselection=1 and d2.dukeselection=1
